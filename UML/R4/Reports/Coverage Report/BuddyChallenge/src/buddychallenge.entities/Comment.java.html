package buddychallenge.entities;

// TODO: Auto-generated Javadoc
/**
 * The Class Comment.
 */
public class Comment {

	/** The id. */
	private static int id;
	
	/** The comment id. */
	private int commentID;
	
	/** The text. */
	private String text;
	
	/** The user. */
	private User user;
	
	/**
	 * Instantiates a new comment.
	 *
	 * @param text the text
	 * @param user the user
	 */
	public Comment(String text, User user) {
		commentID = id++;
		setText(text);
		setUser(user);
	}
	
	/**
	 * Instantiates a new comment.
	 */
	public Comment() {
		this("Nothing.", null);
	}
	
	/**
	 * Gets the comment id.
	 *
	 * @return the comment id
	 */
	public int getCommentID() {
		return commentID;
	}
	
	/**
	 * Gets the text.
	 *
	 * @return the text
	 */
	public String getText() {
		return text;
	}
	
	/**
	 * Sets the text.
	 *
	 * @param text the new text
	 */
	public void setText(String text) {
		this.text = text;
	}
	
	/**
	 * Gets the user.
	 *
	 * @return the user
	 */
	public User getUser() {
		return user;
	}
	
	/**
	 * Sets the user.
	 *
	 * @param user the new user
	 */
	public void setUser(User user) {
		this.user = user;
	}
	
	
	
	
}
